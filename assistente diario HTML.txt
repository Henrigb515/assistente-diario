function formatDate(d) {
  return d.toLocaleDateString('pt-BR', { weekday:'long', day:'numeric', month:'long' });
}

  const d = new Date();

async function callClaude(userMsg, systemMsg, mcpServers) {
  const body = {
    model: "claude-sonnet-4-20250514",
    max_tokens: 1000,
    system: systemMsg || "",
    messages: [{ role: "user", content: userMsg }]
  };
  if (mcpServers) body.mcp_servers = mcpServers;

  const res = await fetch("https://api.anthropic.com/v1/messages", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify(body)
  });
  return res.json();
}

function extractText(data) {
  if (!data || !data.content) return "";
  return data.content.filter(b => b.type === "text").map(b => b.text).join("\n");
}

function extractMCPResult(data) {
  if (!data || !data.content) return null;
  const blocks = data.content.filter(b => b.type === "mcp_tool_result");
  for (const b of blocks) {
    const text = b.content?.[0]?.text || "";
    try { return JSON.parse(text); } catch { return text; }
  }
  return extractText(data);
}

              <div class="weather-desc">${desc}</div>
            </div>
            <div class="weather-extras">
              <div class="weather-row">Sensação <span>${feels}°C</span></div>
              <div class="weather-row">Umidade <span>${humidity}%</span></div>
            </div>
          </div>
        </div>`;
    }

    // Urgent emails
    if (urgent.length > 0) {
      html += `<div class="section"><div class="section-header"><div class="section-dot dot-urgent"></div><div class="section-title">Emails Urgentes / Importantes</div></div>`;
      urgent.forEach((em, i) => {
        html += `<div class="card">
          <div class="card-row">
            <div class="card-left">
              <div class="card-sender">${em.sender}<span class="urgent-badge">URGENTE</span></div>
              <div class="card-subject">${em.subject}</div>
              <div class="card-preview">${em.preview}</div>
              <div class="action-row">
                <span class="action-chip" onclick="openReply('${em.id}','${em.sender?.replace(/'/g,'\\'')}')">↩ Responder</span>
              </div>
            </div>
            <div class="card-time">${em.time}</div>
          </div>
        </div>`;
      });
      html += `</div>`;
    }

    // Unread emails
    if (unread.length > 0) {
      html += `<div class="section"><div class="section-header"><div class="section-dot dot-unread"></div><div class="section-title">Emails Não Lidos</div></div>`;
      unread.forEach((em, i) => {
        html += `<div class="card">
          <div class="card-row">
            <div class="card-left">
              <div class="card-sender">${em.sender}</div>
              <div class="card-subject">${em.subject}</div>
              <div class="card-preview">${em.preview}</div>
              <div class="action-row">
                <span class="action-chip" onclick="openReply('${em.id}','${em.sender?.replace(/'/g,'\\'')}')">↩ Responder</span>
              </div>
            </div>
            <div class="card-time">${em.time}</div>
          </div>
        </div>`;
      });
      html += `</div>`;
    }

    if (urgent.length === 0 && unread.length === 0) {
      html += `<div class="section"><div class="section-header"><div class="section-dot dot-unread"></div><div class="section-title">Emails</div></div><div class="empty-state">Nenhum email novo. Caixa limpa ✓</div></div>`;
    }

    // Today events
    html += `<div class="section"><div class="section-header"><div class="section-dot dot-events"></div><div class="section-title">Compromissos de Hoje</div></div>`;
    if (todayEvents.length > 0) {
      todayEvents.forEach(ev => {
        const t = ev.startTime || '';
        const parts = t.split(':');
        const h = parts[0] || '--';
        const m = parts[1] || '00';
        const hNum = parseInt(h);
        const ampm = isNaN(hNum) ? '' : (hNum >= 12 ? 'PM' : 'AM');
        const h12 = isNaN(hNum) ? h : (hNum > 12 ? hNum - 12 : hNum || 12);
        html += `<div class="event-card">
          <div class="event-time-block">
            <div class="event-hour">${h12}:${m}</div>
            <div class="event-ampm">${ampm}</div>
          </div>
          <div class="event-divider"></div>
          <div class="event-info">
            <div class="event-title">${ev.title}</div>
            <div class="event-detail">${ev.location || 'Sem local definido'} ${ev.endTime ? '· até ' + ev.endTime : ''}</div>
          </div>
        </div>`;
      });
    } else {
      html += `<div class="empty-state">Nenhum evento agendado para hoje.</div>`;
    }
    html += `<div class="action-row" style="margin-top:4px"><span class="action-chip" onclick="openEventModal()">+ Criar evento</span></div></div>`;

    // Tomorrow events
    html += `<div class="section"><div class="section-header"><div class="section-dot dot-tomorrow"></div><div class="section-title">Compromissos de Amanhã</div></div>`;
    if (tomorrowEvents.length > 0) {
      tomorrowEvents.forEach(ev => {
        const t = ev.startTime || '';
        const parts = t.split(':');
        const h = parts[0] || '--';
        const m = parts[1] || '00';
        const hNum = parseInt(h);
        const ampm = isNaN(hNum) ? '' : (hNum >= 12 ? 'PM' : 'AM');
        const h12 = isNaN(hNum) ? h : (hNum > 12 ? hNum - 12 : hNum || 12);
        html += `<div class="event-card">
          <div class="event-time-block">
            <div class="event-hour" style="color:var(--accent)">${h12}:${m}</div>
            <div class="event-ampm">${ampm}</div>
          </div>
          <div class="event-divider"></div>
          <div class="event-info">
            <div class="event-title">${ev.title}</div>
            <div class="event-detail">${ev.location || 'Sem local definido'} ${ev.endTime ? '· até ' + ev.endTime : ''}</div>
          </div>
        </div>`;
      });
    } else {
      html += `<div class="empty-state">Nenhum evento agendado para amanhã.</div>`;
    }
    html += `</div>`;

    sections.innerHTML = html;

  } catch(err) {
    sections.innerHTML = `<div class="empty-state">Erro ao carregar dados. Verifique sua conexão com Gmail e Google Calendar.</div>`;
    console.error(err);
  }
}

function openReply(emailId, sender) {
  currentReplyEmail = { id: emailId, sender };
  document.getElementById('reply-to-name').textContent = sender;
  document.getElementById('reply-textarea').value = '';
  document.getElementById('generating-text').style.display = 'none';
  document.getElementById('reply-modal').classList.add('open');
}

function closeReplyModal() {
  document.getElementById('reply-modal').classList.remove('open');
  currentReplyEmail = null;
}

async function saveDraft() {
  const body = document.getElementById('reply-textarea').value.trim();
  if (!body) return showNotif('Escreva uma resposta antes de salvar.');

  const genEl = document.getElementById('generating-text');
  genEl.style.display = 'block';
  genEl.textContent = 'Salvando rascunho no Gmail...';

  try {
    const email = emailsData.find(e => e.id === currentReplyEmail?.id) || {};
    await callClaude(
      `Crie um rascunho de email respondendo para "${currentReplyEmail?.sender}" com o seguinte conteúdo: "${body}". Email original: assunto "${email.subject || ''}"`,
      "Você é um assistente de email. Use a ferramenta do Gmail para criar o rascunho.",
      [{ type:"url", url: GMAIL_MCP, name:"gmail" }]
    );
    showNotif('✓ Rascunho salvo no Gmail!');
    closeReplyModal();
  } catch(e) {
    showNotif('Erro ao salvar rascunho.');
    genEl.style.display = 'none';
  }
}

async function createEvent() {
  const title = document.getElementById('ev-title').value.trim();
  const date = document.getElementById('ev-date').value;
  const time = document.getElementById('ev-time').value;
  const location = document.getElementById('ev-location').value.trim();

  if (!title || !date || !time) return showNotif('Preencha título, data e horário.');

  try {
    await callClaude(
      `Crie um evento no Google Calendar: título "${title}", data ${date}, horário ${time}${location ? ', local: ' + location : ''}.`,
      "Você é assistente de calendário. Use a ferramenta para criar o evento.",
      [{ type:"url", url: GCAL_MCP, name:"gcal" }]
    );
    showNotif(`✓ Evento "${title}" criado!`);
    closeEventModal();
    loadAll();
  } catch(e) {
    showNotif('Erro ao criar evento.');
  }
}

async function handleUserInput(text) {
  if (!text.trim()) return;
  const lower = text.toLowerCase();

  if (lower.includes('evento') || lower.includes('agendar') || lower.includes('reunião') || lower.includes('marcar')) {
    await callClaude(
      `O usuário pediu: "${text}". Crie o evento no Google Calendar conforme o pedido.`,
      "Você é assistente de calendário. Use a ferramenta para criar o evento.",
      [{ type:"url", url: GCAL_MCP, name:"gcal" }]
    );
    showNotif('✓ Evento criado a partir do seu pedido!');
    loadAll();
  } else if (lower.includes('responder') || lower.includes('email') || lower.includes('reply')) {
    const parts = emailsData.slice(0,3);
    if (parts.length > 0) openReply(parts[0].id, parts[0].sender);
    else showNotif('Nenhum email recente encontrado.');
  } else if (lower.includes('atualizar') || lower.includes('recarregar') || lower.includes('bom dia')) {
    loadAll();
  } else {
    const res = await callClaude(
      `Contexto: assistente pessoal do usuário. Emails recentes: ${JSON.stringify(emailsData.slice(0,3))}. Eventos hoje: ${JSON.stringify(eventsData)}. Pergunta: "${text}"`,
      "Você é um assistente pessoal prestativo. Responda de forma concisa em português.",
      [{ type:"url", url: GMAIL_MCP, name:"gmail" }, { type:"url", url: GCAL_MCP, name:"gcal" }]
    );
    showNotif(extractText(res).slice(0, 80) + '...');
  }
}

// Mic (Web Speech API)
let recognition = null;
const micBtn = document.getElementById('mic-btn');
let listening = false;

if ('webkitSpeechRecognition' in window || 'SpeechRecognition' in window) {
  const SR = window.SpeechRecognition || window.webkitSpeechRecognition;
  recognition = new SR();
  recognition.lang = 'pt-BR';
  recognition.continuous = false;
  recognition.interimResults = false;

  recognition.onresult = e => {
    const transcript = e.results[0][0].transcript;
    document.getElementById('user-input').value = transcript;
    handleUserInput(transcript);
    listening = false;
    micBtn.style.background = 'var(--accent)';
  };

  recognition.onerror = () => {
    listening = false;
    micBtn.style.background = 'var(--accent)';
  };

  recognition.onend = () => {
    listening = false;
    micBtn.style.background = 'var(--accent)';
  };

  micBtn.addEventListener('click', () => {
    if (!listening) {
      recognition.start();
      listening = true;
      micBtn.style.background = 'var(--urgent)';
    } else {
      recognition.stop();
      listening = false;
      micBtn.style.background = 'var(--accent)';
    }
  });
} else {
  micBtn.title = 'Microfone não suportado neste browser';
}
